```python
"""Download and cache external feature data from yfinance and FRED."""
import argparse
import time
from datetime import datetime
from pathlib import Path

import pandas as pd
import yfinance as yf

DATA_DIR = Path(__file__).parent / "data"

# yfinance tickers
YF_TICKERS = {
    "vix_close": "^VIX",
    "vix3m_close": "^VIX3M",
    "hyg_close": "HYG",
    "lqd_close": "LQD",
    "gld_close": "GLD",
    "uup_close": "UUP",
    "tlt_close": "TLT",
    "eem_close": "EEM",
}

# FRED series (public CSV endpoint, no API key needed)
FRED_SERIES = {
    "dgs10": "DGS10",
    "dgs2": "DGS2",
}

FRED_URL = "https://fred.stlouisfed.org/graph/fredgraph.csv?id={series}&cosd={start}&coed={end}"


def download_yfinance(start: str, end: str) -> pd.DataFrame:
    """Download close prices for all yfinance tickers."""
    frames = {}
    for col_name, ticker in YF_TICKERS.items():
        print(f"  Downloading {ticker}...")
        data = yf.download(ticker, start=start, end=end, auto_adjust=True, progress=False)
        if len(data) == 0:
            print(f"  WARNING: No data for {ticker}")
            continue
        # Handle multi-level columns from yfinance
        if isinstance(data.columns, pd.MultiIndex):
            data.columns = data.columns.get_level_values(0)
        frames[col_name] = data["Close"].rename(col_name)
        time.sleep(0.5)  # rate limit

    return pd.concat(frames.values(), axis=1).sort_index()


def download_fred(start: str, end: str) -> pd.DataFrame:
    """Download yield data from FRED public CSV endpoint."""
    frames = {}
    for col_name, series in FRED_SERIES.items():
        url = FRED_URL.format(series=series, start=start, end=end)
        print(f"  Downloading FRED {series}...")
        df = pd.read_csv(url, parse_dates=["DATE"], index_col="DATE", na_values=".")
        df.columns = [col_name]
        # Forward-fill gaps (max 5 business days)
        df = df.ffill(limit=5)
        frames[col_name] = df[col_name]

    return pd.concat(frames.values(), axis=1).sort_index()


def download_all(start: str = "1993-01-01", end: str | None = None, force: bool = False) -> Path:
    """Download all external feature data and save to CSV."""
    DATA_DIR.mkdir(parents=True, exist_ok=True)
    output = DATA_DIR / "features_daily.csv"

    # Idempotency: skip if recent file exists
    if not force and output.exists():
        age_hours = (time.time() - output.stat().st_mtime) / 3600
        if age_hours < 24:
            print(f"Using cached data ({age_hours:.1f}h old). Use --force to re-download.")
            return output

    if end is None:
        end = datetime.now().strftime("%Y-%m-%d")

    print("Downloading yfinance data...")
    yf_data = download_yfinance(start, end)

    print("Downloading FRED data...")
    fred_data = download_fred(start, end)

    # Merge on date (inner join — only keep days where all sources have data)
    combined = yf_data.join(fred_data, how="inner")
    combined = combined.dropna()
    combined.index.name = "date"

    combined.to_csv(output)
    print(f"Saved {len(combined)} rows to {output}")
    return output


if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Download external feature data")
    parser.add_argument("--force", action="store_true", help="Force re-download even if cache exists")
    args = parser.parse_args()
    download_all(force=args.force)
```

```python
    def compute_features_extended(
        self, spy_df: pd.DataFrame, external_df: pd.DataFrame,
    ) -> pd.DataFrame:
        """Compute all candidate features (base + external) for feature selection.

        Args:
            spy_df: DataFrame with open, high, low, close, volume columns.
            external_df: DataFrame with external data columns (vix_close, etc.).

        Returns:
            DataFrame with all 14 candidate feature columns, NaN rows dropped.
        """
        result = spy_df.copy()

        # Base SPY features (same as compute_features)
        result["log_return_5d"] = np.log(result["close"] / result["close"].shift(5))
        result["log_return_20d"] = np.log(result["close"] / result["close"].shift(20))
        result["volatility_20d"] = result["close"].pct_change().rolling(20).std()

        # Join external data
        ext = external_df.reindex(result.index)

        # VIX features
        result["vix_level"] = ext["vix_close"]
        result["vix_term_structure"] = ext["vix_close"] / ext["vix3m_close"]

        # Credit
        result["credit_spread"] = np.log(ext["hyg_close"] / ext["lqd_close"])
        result["credit_spread"] = result["credit_spread"] - result["credit_spread"].shift(20)

        # Rates
        result["yield_curve_slope"] = ext["dgs10"] - ext["dgs2"]
        result["yield_10y_level"] = ext["dgs10"]

        # Cross-asset momentum (20d log returns)
        result["gold_momentum"] = np.log(ext["gld_close"] / ext["gld_close"].shift(20))
        result["dollar_momentum"] = np.log(ext["uup_close"] / ext["uup_close"].shift(20))
        result["bond_momentum"] = np.log(ext["tlt_close"] / ext["tlt_close"].shift(20))
        result["em_momentum"] = np.log(ext["eem_close"] / ext["eem_close"].shift(20))

        # Derived from SPY
        result["vol_of_vol"] = result["volatility_20d"].rolling(20).std()
        result["drawdown_depth"] = result["close"] / result["close"].rolling(252).max() - 1

        # Drop NaN rows
        result = result.dropna(subset=EXTENDED_FEATURE_COLUMNS)

        return result

    def compute_features_with_labels_extended(
        self, spy_df: pd.DataFrame, external_df: pd.DataFrame,
    ) -> pd.DataFrame:
        """Compute extended features + forward-looking label for training."""
        result = self.compute_features_extended(spy_df, external_df)

        forward_return = result["close"].shift(-5) / result["close"] - 1
        result = result[forward_return.notna()].copy()
        forward_return = forward_return[forward_return.notna()]

        result["forward_return_5d_class"] = CLASS_FLAT
        result.loc[forward_return > RETURN_THRESHOLD, "forward_return_5d_class"] = CLASS_UP
        result.loc[forward_return < -RETURN_THRESHOLD, "forward_return_5d_class"] = CLASS_DOWN
        result["forward_return_5d_class"] = result["forward_return_5d_class"].astype(int)

        return result
```

### Task 3: Extract Reusable Walk-Forward Evaluator

**Files:**

The current `train_model.py` has walk-forward logic mixed with reporting. We need a reusable function that takes a feature list, runs walk-forward, and returns Sharpe ratios. This also fixes the Sharpe calculation bug (computing returns within each window, not across concatenated dates).

- [ ] **Step 1: Create walk_forward.py**

Create `backtest/walk_forward.py`:

```python
"""Reusable walk-forward evaluation for feature selection."""
import numpy as np
import pandas as pd

from schroeder_trader.config import SLIPPAGE_ESTIMATE
from schroeder_trader.strategy.feature_engineer import CLASS_DOWN, CLASS_UP
from schroeder_trader.strategy.xgboost_classifier import train_model

# Walk-forward parameters
TRAIN_YEARS = 2
TEST_MONTHS = 6


def evaluate_features(
    features_df: pd.DataFrame,
    feature_columns: list[str],
    verbose: bool = False,
) -> dict:
    """Run walk-forward validation for a given feature set.

    Args:
        features_df: DataFrame with all feature columns and 'forward_return_5d_class' label
                     and 'close' price column.
        feature_columns: List of feature column names to use.
        verbose: Print per-window progress.

    Returns:
        Dict with 'full_sharpe', 'post_2020_sharpe', 'accuracy', 'n_windows'.
    """
    all_window_returns = []
    all_window_dates = []
    all_predictions = []
    all_actuals = []

    start_date = features_df.index.min()
    end_date = features_df.index.max()
    train_start = start_date
    n_windows = 0

    while True:
        train_end = train_start + pd.DateOffset(years=TRAIN_YEARS)
        test_end = train_end + pd.DateOffset(months=TEST_MONTHS)

        if train_end >= end_date:
            break

        train_data = features_df[(features_df.index >= train_start) & (features_df.index < train_end)]
        test_data = features_df[(features_df.index >= train_end) & (features_df.index < test_end)]

        # Check all required features are non-NaN in this window
        train_valid = train_data.dropna(subset=feature_columns)
        test_valid = test_data.dropna(subset=feature_columns)

        if len(train_valid) < 100 or len(test_valid) < 10:
            train_start = train_start + pd.DateOffset(months=TEST_MONTHS)
            continue

        X_train = train_valid[feature_columns]
        y_train = train_valid["forward_return_5d_class"]
        X_test = test_valid[feature_columns]
        y_test = test_valid["forward_return_5d_class"]

        # Early stopping validation split
        val_split = int(len(X_train) * 0.8)
        model = train_model(X_train[:val_split], y_train[:val_split],
                            X_train[val_split:], y_train[val_split:])

        preds = model.predict(X_test)

        # Compute strategy returns WITHIN this window (avoids cross-window gap bug)
        close = test_valid["close"]
        window_returns = close.pct_change().fillna(0).values

        position = 0
        strat_returns = []
        for i, (pred, ret) in enumerate(zip(preds, window_returns)):
            prev_position = position
            if pred == CLASS_UP:
                position = 1
            elif pred == CLASS_DOWN:
                position = 0

            strat_ret = position * ret
            if position != prev_position:
                strat_ret -= SLIPPAGE_ESTIMATE
            strat_returns.append(strat_ret)

        all_window_returns.extend(strat_returns)
        all_window_dates.extend(test_valid.index)
        all_predictions.extend(preds)
        all_actuals.extend(y_test.values)
        n_windows += 1

        if verbose:
            from sklearn.metrics import accuracy_score
            acc = accuracy_score(y_test, preds)
            print(f"  Window {n_windows}: {train_start.date()}-{train_end.date()} → "
                  f"{train_end.date()}-{test_end.date()}, acc={acc:.4f}")

        train_start = train_start + pd.DateOffset(months=TEST_MONTHS)

    if len(all_window_returns) == 0:
        return {"full_sharpe": 0.0, "post_2020_sharpe": 0.0, "accuracy": 0.0, "n_windows": 0}

    returns = np.array(all_window_returns)
    dates = pd.DatetimeIndex(all_window_dates)
    preds = np.array(all_predictions)
    actuals = np.array(all_actuals)

    full_sharpe = _compute_sharpe(returns)

    post_2020_mask = dates >= "2020-01-01"
    post_sharpe = _compute_sharpe(returns[post_2020_mask]) if post_2020_mask.any() else 0.0

    from sklearn.metrics import accuracy_score
    accuracy = accuracy_score(actuals, preds)

    return {
        "full_sharpe": round(full_sharpe, 4),
        "post_2020_sharpe": round(post_sharpe, 4),
        "accuracy": round(accuracy, 4),
        "n_windows": n_windows,
    }


def _compute_sharpe(returns: np.ndarray) -> float:
    """Compute annualized Sharpe ratio."""
    if len(returns) == 0 or returns.std() < 1e-10:
        return 0.0
    return float(np.sqrt(252) * returns.mean() / returns.std())
```

```python
"""Forward feature selection using walk-forward Sharpe as scoring metric."""
import json
import time
from pathlib import Path

import pandas as pd

from backtest.train_model import load_data
from backtest.walk_forward import evaluate_features
from schroeder_trader.strategy.feature_engineer import (
    FeaturePipeline,
    EXTENDED_FEATURE_COLUMNS,
)

DATA_DIR = Path(__file__).parent / "data"
OUTPUT_DIR = Path(__file__).parent / "results"

# Base features (SHAP-validated from Phase 2)
BASE_FEATURES = ["log_return_5d", "log_return_20d", "volatility_20d"]

# Candidates to evaluate (extended minus base)
CANDIDATE_FEATURES = [f for f in EXTENDED_FEATURE_COLUMNS if f not in BASE_FEATURES]

# Selection thresholds
MIN_SHARPE_IMPROVEMENT = 0.02
TARGET_FULL_SHARPE = 0.88
TARGET_POST_2020_SHARPE = 1.14


def load_extended_features() -> pd.DataFrame:
    """Load SPY + external data and compute all features with labels."""
    spy_df = load_data()

    ext_path = DATA_DIR / "features_daily.csv"
    if not ext_path.exists():
        raise FileNotFoundError(
            f"No external features at {ext_path}. Run download_features.py first."
        )
    ext_df = pd.read_csv(ext_path, index_col="date", parse_dates=True)

    pipeline = FeaturePipeline()
    return pipeline.compute_features_with_labels_extended(spy_df, ext_df)


def run_selection() -> dict:
    """Run greedy forward feature selection."""
    print("Loading and computing features...")
    features_df = load_extended_features()
    print(f"Feature matrix: {len(features_df)} rows, date range: "
          f"{features_df.index.min().date()} to {features_df.index.max().date()}")

    selected = BASE_FEATURES.copy()
    candidates = CANDIDATE_FEATURES.copy()

    # Evaluate baseline
    print(f"\nStep 0: Evaluating base features {selected}...")
    baseline = evaluate_features(features_df, selected, verbose=False)
    current_sharpe = baseline["full_sharpe"]
    current_post2020 = baseline["post_2020_sharpe"]

    steps = []
    print(f"\n{'Step':<6}{'Added Feature':<25}{'Full Sharpe':<14}{'Post-2020':<14}{'Delta':<10}")
    print("-" * 69)
    print(f"{'0':<6}{'(base)':<25}{current_sharpe:<14.4f}{current_post2020:<14.4f}{'—':<10}")

    step = 0
    while candidates:
        step += 1
        best_feature = None
        best_sharpe = current_sharpe
        best_post2020 = current_post2020

        for i, candidate in enumerate(candidates):
            test_set = selected + [candidate]
            print(f"  Testing {candidate} ({i+1}/{len(candidates)})...", end="", flush=True)
            t0 = time.time()
            result = evaluate_features(features_df, test_set, verbose=False)
            elapsed = time.time() - t0
            print(f" Sharpe={result['full_sharpe']:.4f} ({elapsed:.1f}s)")

            if (result["full_sharpe"] > best_sharpe + MIN_SHARPE_IMPROVEMENT
                    and result["post_2020_sharpe"] >= current_post2020):
                best_feature = candidate
                best_sharpe = result["full_sharpe"]
                best_post2020 = result["post_2020_sharpe"]

        if best_feature is None:
            print(f"\nStep {step}: No candidate improves Sharpe by >= {MIN_SHARPE_IMPROVEMENT}. Stopping.")
            break

        selected.append(best_feature)
        candidates.remove(best_feature)
        delta = best_sharpe - current_sharpe
        current_sharpe = best_sharpe
        current_post2020 = best_post2020

        steps.append({
            "step": step,
            "added_feature": best_feature,
            "full_sharpe": current_sharpe,
            "post_2020_sharpe": current_post2020,
            "delta": round(delta, 4),
        })

        print(f"\n{step:<6}{best_feature:<25}{current_sharpe:<14.4f}{current_post2020:<14.4f}{f'+{delta:.4f}':<10}")

        if current_sharpe >= TARGET_FULL_SHARPE and current_post2020 >= TARGET_POST_2020_SHARPE:
            print(f"\nBoth targets met! Full={current_sharpe:.4f} >= {TARGET_FULL_SHARPE}, "
                  f"Post-2020={current_post2020:.4f} >= {TARGET_POST_2020_SHARPE}")
            break

    # Summary
    targets_met = current_sharpe >= TARGET_FULL_SHARPE and current_post2020 >= TARGET_POST_2020_SHARPE

        "final_full_sharpe": current_sharpe,
        "final_post_2020_sharpe": current_post2020,
        "targets_met": targets_met,
    }

    OUTPUT_DIR.mkdir(parents=True, exist_ok=True)

        json.dump(results, f, indent=2)
    print(f"\nResults saved to {OUTPUT_DIR / 'feature_selection_results.json'}")
```
